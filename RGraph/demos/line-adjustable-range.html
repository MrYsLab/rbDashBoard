An adjustable range Line chart

An adjustable range chart.

[No canvas support]

This goes in the documents header (or you could place it just above the jQuery ready event code): 
<script src="http://ajax.googleapis.com/ajax/libs/jquery/1.10.2/jquery.min.js"></script>
<script src="RGraph.common.core.js"></script>
<script src="RGraph.common.dynamic.js"></script>
<script src="RGraph.line.js"></script>
Put this where you want the chart to show up: 
<canvas id="cvs" width="600" height="250" !style="border: 1px solid #ccc">
    [No canvas support]
</canvas>
This is the code that generates the chart. Because it's using the jQuery ready event you can put this at the bottom of the document: 
<script>
    $(document).ready(function ()
    {
        var l1 = new RGraph.Line({
            id: 'cvs',
            data: [
                   [8,8,6,6,8,9,13,15,16,15,15,12,11,11,13,10,12,11,10,13,12,9,8,11,12,10,16,15,12,11],
                   [4,5,3,2,5,6,8,9,5,6,4,5,3,2,5,6,8,9,5,6,4,5,3,2,5,6,8,9,5,6]
                  ],
            options: {
                'gutter.top': 50,
                labels: ['13th','14th','15th','16th','15th','16th','17th','18th','19th'],
                title: 'Range of fuel consumption',
                background: {
                    grid: {
                        autofit: {
                            align: true
                        }
                    }
                },
                filled: {
                    self: true,
                    range: true
                },
                fillstyle: 'rgba(128,255,128,0.5)',
                linewidth: 2,
                colors: ['green'],
                numxticks: 8,
                adjustable: true,
                tickmarks: null,
                ymax: 25,
                'text.size': 14
            }
        }).draw();
    });
</script>
