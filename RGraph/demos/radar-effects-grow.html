An animated Radar chart

[No canvas support]

This goes in the documents header (or you could place it just above the jQuery ready event code): 
<script src="http://ajax.googleapis.com/ajax/libs/jquery/1.10.2/jquery.min.js"></script>
<script src="RGraph.common.core.js"></script>
<script src="RGraph.common.dynamic.js"></script>
<script src="RGraph.common.tooltips.js"></script>
<script src="RGraph.common.effects.js"></script>
<script src="RGraph.radar.js"></script>
Put this where you want the chart to show up: 
<canvas id="cvs" width="450" height="400">
    [No canvas support]
</canvas>
This is the code that generates the chart. Because it's using the jQuery ready event you can put this at the bottom of the document: 
<script>
    $(document).ready(function ()
    {
        var radar2 = new RGraph.Radar({
            id: 'cvs',
            data: [
                   [4,8,5,2,3],
                   [9,5,6,3,5]
                  ],
            options: {
                labels: ['Mal', 'Barry', 'Gary','Dave','Paul'],
                tooltips: ['Mal', 'Barry', 'Gary','Dave','Paul', 'Mal', 'Barry', 'Gary','Dave','Paul'],
                clearto: 'white',
                background: {
                    circles: {
                        poly: true,
                        spacing: 30
                    }
                },
                colors: ['rgba(255,0,0,0.25)','rgba(255,255,0,0.25)'],
                axes: {
                    color: 'transparent'
                },
                highlights: true,
                strokestyle: ['red','black']
            }
        }).on('beforedraw', function (obj)
        {
            RGraph.clear(obj.canvas, 'white');
        }).grow();
    });
</script>
