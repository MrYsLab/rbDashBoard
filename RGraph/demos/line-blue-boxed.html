A Line chart in blue

A Line chart in blue surrounded by a black box. The box is added after the main drawing by using the ondraw custom RGraph event.

[No canvas support]

This goes in the documents header (or you could place it just above the jQuery ready event code): 
<script src="http://ajax.googleapis.com/ajax/libs/jquery/1.10.2/jquery.min.js"></script>
<script src="RGraph.common.core.js"></script>
<script src="RGraph.line.js"></script>
Put this where you want the chart to show up: 
<canvas id="cvs" width="600" height="250">
    [No canvas support]
</canvas>
This is the code that generates the chart. Because it's using the jQuery ready event you can put this at the bottom of the document: 
<script>
    $(document).ready(function ()
    {
        var line = new RGraph.Line({
            id: 'cvs',
            data: [28,21,26,32,35,36,39,45,47,51,55,74,85],
            options: {
                labels: ['1950','1955','1960','1965','1970','1975','1980','1985','1990','1995','2000','2005','2010'],
                colors: ['blue'],
                noaxes: true,
                ymax: 100,
                units: {
                    post: '%'
                },
                gutter: {
                    left: 45,
                    bottom: 45,
                    top: 10,
                    right: 5
                },
                text: {
                    angle: 45
                },
                linewidth: 3,
                hmargin: 7,
                background: {
                    grid: {
                        autofit: {
                            numvlines: 12
                        }
                    }
                }
            }
        }).on('draw',function (obj)
        {
            var co = obj.context;
            
            co.lineWidth = 1;
            co.lineJoin = 'miter';
            co.strokeRect(45,10,550.01,195); // The .01 is to get around a Chrome oddity
        })
        .draw()
    });
</script>
