A Meter chart customised to look a bit like the Google charts Meter

[No canvas support]

This goes in the documents header (or you could place it just above the jQuery ready event code): 
<script src="http://ajax.googleapis.com/ajax/libs/jquery/1.10.2/jquery.min.js"></script>
<script src="RGraph.common.core.js"></script>
<script src="RGraph.common.dynamic.js"></script>
<script src="RGraph.meter.js"></script>
Put this where you want the chart to show up: 
<canvas id="cvs" width="600" height="250" style="border-radius: 15px">
    [No canvas support]
</canvas>
This is the code that generates the chart. Because it's using the jQuery ready event you can put this at the bottom of the document: 
<script>
    $(document).ready(function ()
    {
        var meter = new RGraph.Meter({
            id: 'cvs',
            min: 0,
            max: 100,
            value: 75,
            options: {
                centerpin: {
                    stroke: 'rgba(0,0,0,0)',
                    fill: 'rgba(0,0,0,0)'
                },
                colors: {
                    ranges: [
                        [0,10,'#f20'],
                        [10,20,'#f30'],
                        [20,30,'#f50'],
                        [30,40,'#f60'],
                        [40,50,'#f80'],
                        [50,60,'#fa0'],
                        [60,70,'#fc0'],
                        [70,80,'#fd0'],
                        [80,90,'#ff0'],
                        [90,100,'#ff0'],
                    ]
                },
                labels: {
                    count: 0
                },
                angles: {
                    start: RGraph.PI + 0.5,
                    end: RGraph.TWOPI - 0.5
                },
                linewidth: {
                    segments: 0
                },
                text: {
                    size: 16
                },
                strokestyle: 'white',
                segment: {
                    radius: {
                        start: 150  
                    }
                },
                needle: {
                    radius: 210
                },
                border: 0,
                tickmarks: {
                    small: {
                        num: 0
                    },
                    big: {
                        num: 0
                    }
                },
                adjustable: true
            }
        }).draw()
    });
</script>
