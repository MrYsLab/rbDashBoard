Two charts (on two canvas tags) with tooltips and click events

[No canvas support] [No canvas support]

This goes in the documents header (or you could place it just above the jQuery ready event code): 
<script src="http://ajax.googleapis.com/ajax/libs/jquery/1.10.2/jquery.min.js"></script>
<script src="RGraph.common.core.js"></script>
<script src="RGraph.common.dynamic.js"></script>
<script src="RGraph.common.tooltips.js"></script>
<script src="RGraph.line.js"></script>
<script src="RGraph.pie.js"></script>
Put this where you want the chart to show up: 
<canvas id="cvs" width="600" height="250">[No canvas support]</canvas>
<canvas id="cvs2" width="200" height="200">[No canvas support]</canvas>
This is the code that generates the chart. Because it's using the jQuery ready event you can put this at the bottom of the document: 
<script>
    $(document).ready(function ()
    {
        var line = new RGraph.Line({
            id: 'cvs',
            data: [20,30,50,40,30,50,40,30,60,50,10,20],
            options: {
                labels: ['Jan','Feb','Mar','Apr','May','Jun','Jul','Aug','Sep','Oct','Nov','Dec'],
                tooltips: ['Jan','Feb','Mar','Apr','May','Jun','Jul','Aug','Sep','Oct','Nov','Dec'],
                events: {
                    click: function (e, shape) {alert('The Line chart click event');}
                }
            }
        }).draw()
        
        var pie = new RGraph.Pie({
            id: 'cvs2',
            data: [5,6],
            options: {
                exploded: 5,
                strokestyle: 'rgba(0,0,0,0)',
                events: {
                    click: function (e, shape) {alert('The Pie chart click event');}
                },
                tooltips: ['John', 'Mary']
            }
        }).draw();
    });
</script>
