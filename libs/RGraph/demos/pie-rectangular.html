A "rectangular" Pie chart

By using a large chart.radius setting the Pie will be too large for the canvas and you'll get a rectangular Pie chart effect.

[No canvas support]

This goes in the documents header (or you could place it just above the jQuery ready event code): 
<script src="http://ajax.googleapis.com/ajax/libs/jquery/1.10.2/jquery.min.js"></script>
<script src="RGraph.common.core.js"></script>
<script src="RGraph.common.key.js"></script>
<script src="RGraph.common.effects.js"></script>
<script src="RGraph.pie.js"></script>
Put this where you want the chart to show up: 
<canvas id="cvs" width="350" height="250">
    [No canvas support]
</canvas>
This is the code that generates the chart. Because it's using the jQuery ready event you can put this at the bottom of the document: 
<script>
    $(document).ready(function ()
    {
        var pie = new RGraph.Pie({
            id: 'cvs',
            data: [4,8,4,6,5,3,2,5],
            options: {
                radius: 350,
                key: ['Alvin','Pete','Hoolio','Jack','Kev','Luis','Lou','Jesse'],
                strokestyle: 'white',
                linewidth: 3
            }
        }).draw()
        
        if (RGraph.ISIE9UP || RGraph.ISCHROME) {
            pie.roundRobin(null,addCenterPin);
            pie.implode();
        } else if (RGraph.ISOLD) {
            pie.draw()
        } else {
            pie.grow();
        }
        
        function addCenterPin ()
        {
            RGraph.redraw();
        }
    });
</script>
