An inverted Funnel chart showing a hypothetical company structure

[No canvas support]

This goes in the documents header (or you could place it just above the jQuery ready event code): 
<script src="http://ajax.googleapis.com/ajax/libs/jquery/1.10.2/jquery.min.js"></script>
<script src="RGraph.common.core.js"></script>
<script src="RGraph.funnel.js"></script>
Put this where you want the chart to show up: 
<canvas id="cvs" width="450" height="500">
    [No canvas support]
</canvas>
This is the code that generates the chart. Because it's using the jQuery ready event you can put this at the bottom of the document: 
<script>
    $(document).ready(function ()
    {
        var funnel = new RGraph.Funnel({
            id: 'cvs',
            data: [15,26,54,84,100, 186],
            options: {
                labels: {
                    self: ['Directors','Regional directors', 'Regional Managers','Managers','Supervisors','Programmers'],
                    sticks: true,
                    x: 5
                },
                gutter: {
                    left: 230
                },
                strokestyle: 'rgba(0,0,0,0)',
                text: {
                    boxed: false
                },
                shadow: {
                    self: true,
                    offsetx: 0,
                    offsety: 0,
                    blur: 15,
                    color: 'gray'
                },
                colors: ['red','green','blue','pink','#666']
            }
        }).draw();
    });
</script>
