A Pie chart with the key in gutter mode

[No canvas support]

This goes in the documents header (or you could place it just above the jQuery ready event code): 
<script src="http://ajax.googleapis.com/ajax/libs/jquery/1.10.2/jquery.min.js"></script>
<script src="RGraph.common.core.js"></script>
<script src="RGraph.common.key.js"></script>
<script src="RGraph.common.dynamic.js"></script>
<script src="RGraph.drawing.rect.js"></script>
<script src="RGraph.pie.js"></script>
Put this where you want the chart to show up: 
<canvas id="cvs" width="350" height="350" style="border: 1px solid gray">
    [No canvas support]
</canvas>
This is the code that generates the chart. Because it's using the jQuery ready event you can put this at the bottom of the document: 
<script>
    $(document).ready(function ()
    {
        var pie = new RGraph.Pie({
            id: 'cvs',
            data: [45,43,23,68,84,41],
            options: {
                title: {
                    self: 'A pie chart showing gutter mode',
                    y: 30
                },
                key: {
                    self: ['John','Fred','Lou','Pete','Kevin','Gary'],
                    colors: ['red','#ccc','#0f0','#00f','pink','yellow'],
                    interactive: true,
                    rounded: false,
                    position: {
                        self: 'gutter',
                        y: 315
                    }
                },
                radius: 125,
                linewidth: 1
            }
        }).draw();
    });
</script>
