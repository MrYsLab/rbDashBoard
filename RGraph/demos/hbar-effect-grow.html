A Horizontal Bar chart example

[No canvas support]

This goes in the documents header (or you could place it just above the jQuery ready event code): 
<script src="http://ajax.googleapis.com/ajax/libs/jquery/1.10.2/jquery.min.js"></script>
<script src="RGraph.common.core.js"></script>
<script src="RGraph.common.effects.js"></script>
<script src="RGraph.hbar.js"></script>
Put this where you want the chart to show up: 
<canvas id="cvs" width="600" height="400">
    [No canvas support]
</canvas>
This is the code that generates the chart. Because it's using the jQuery ready event you can put this at the bottom of the document: 
<script>
    $(document).ready(function ()
    {
        var hbar = new RGraph.HBar({
            id: 'cvs',
            data: [0.1,0.2,0.3,3.2,0.2,0.5,0.5,0.2,0.4,0.8,0,0.1],
            options: {
                background: {
                    grid: false
                },
                xmax: 3.5,
                scale: {
                    decimals: 1
                },
                colors: {
                    self: ['#164366','#164366','#164366','#FDB515','#164366'],
                    sequential: true
                },
                labels: {
                    self: ['Norway','Australia','Netherlands','United States','New Zealand','Canada','Ireland','Germany','Sweden','Switzerland','Japan','England and Wales'],
                    above: {
                        self: true,
                        decimals: 1
                    }
                },
                noxaxis: true,
                gutter: {
                    left: 125
                },
                xlabels: false
            }
        }).grow();
    });
</script>
