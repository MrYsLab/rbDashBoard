A grouped bar chart with background image

[No canvas support]

This goes in the documents header (or you could place it just above the jQuery ready event code): 
<script src="http://ajax.googleapis.com/ajax/libs/jquery/1.10.2/jquery.min.js"></script>
<script src="RGraph.common.core.js"></script>
<script src="RGraph.common.effects.js"></script>
<script src="RGraph.common.key.js"></script>
<script src="RGraph.bar.js"></script>
Put this where you want the chart to show up: 
<canvas id="cvs" width="600" height="250" !style="border: 1px solid #ccc">
    [No canvas support]
</canvas>
This is the code that generates the chart. Because it's using the jQuery ready event you can put this at the bottom of the document: 
<script>
    $(document).ready(function ()
    {
        var bar = new RGraph.Bar({
            id: 'cvs',
            data: [[5,6],[7, 7],[6, 9],[8, 12],[12, 13],[15, 18],[45, 42],[55, 53],[36, 37],[25, 28],[20, 24],[15, 18]],
            options: {
                background: {
                    image: '../images/bg.png',
                    grid: false
                },
                labels: ['Jan','Feb','Mar','Apr','May','Jun','Jul','Aug','Sep','Oct','Nov','Dec'],
                key: {
                    self: ['Sunshine in 2000', 'Sunshine in 2005'],
                    position: {
                        self: 'gutter',
                        gutter: {
                            boxed: false
                        }
                    },
                    background: 'rgb(255,255,255)'
                },
                colors: ['yellow', 'red'],
                shadow: {
                    self: true, // Defaults to true anyway
                    blur: 15,
                    offsetx: 0,
                    offsety: 0,
                    color: '#aaa'
                },
                strokestyle: 'rgba(0,0,0,0)',
                gutter: {
                    left: 55,
                    right: 5
                },
                hmargin: {
                    grouped: 1
                }
            }
        }).fadeIn();
    });
</script>
