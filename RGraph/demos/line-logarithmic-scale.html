A demo of a logarithmic chart

This demonstrates a logarithmic chart using log(10). It has custom Y labels.

[No canvas support]

This goes in the documents header (or you could place it just above the jQuery ready event code): 
<script src="http://ajax.googleapis.com/ajax/libs/jquery/1.10.2/jquery.min.js"></script>
<script src="RGraph.common.core.js"></script>
<script src="RGraph.common.dynamic.js"></script>
<script src="RGraph.common.tooltips.js"></script>
<script src="RGraph.common.drawing.rect.js"></script>
<script src="RGraph.line.js"></script>
Put this where you want the chart to show up: 
<canvas id="cvs" width="600" height="250">
    [No canvas support]
</canvas>
This is the code that generates the chart. Because it's using the jQuery ready event you can put this at the bottom of the document: 
<script>
    $(document).ready(function ()
    {
        var data = [999999,5,6,4,6,8012,12,7];

        for (var i=0;i<data.length; ++i) {
            data[i] = RGraph.log(data[i], 10); // This function is in RGraph.common.core.js
        }




        var line = new RGraph.Line({id: 'cvs', data: data})

        line.set({
            numyticks: 6,
            ymax: 6,
            ylabels: {
                specific: [
                    RGraph.numberFormat(line, 1000000),
                    RGraph.numberFormat(line, 100000),
                    RGraph.numberFormat(line, 10000),
                    RGraph.numberFormat(line, 1000),
                    100, // No need to format this
                    10,   // No need to format this
                    ''
                ]
            },
            'text.size': 14,
            'gutter.left': 95,
            labels: ['Pob','Libby','June','Hoolio','Jane','Daz','Luis','John'],
            numxticks: 7,
            background: {
                grid: {
                    autofit: {
                        numhlines: 6,
                        numvlines: 7
                    }
                }
            }
        }).draw();
    });
</script>
