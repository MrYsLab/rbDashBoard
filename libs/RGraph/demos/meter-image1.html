A sketchy (interactive) Meter chart with background and needle images

[No canvas support]

This goes in the documents header (or you could place it just above the jQuery ready event code): 
<script src="http://ajax.googleapis.com/ajax/libs/jquery/1.10.2/jquery.min.js"></script>
<script src="RGraph.common.core.js"></script>
<script src="RGraph.common.dynamic.js"></script>
<script src="RGraph.meter.js"></script>
Put this where you want the chart to show up: 
<canvas id="cvs" id="cvs" width="237" height="196">
    [No canvas support]
</canvas>
This is the code that generates the chart. Because it's using the jQuery ready event you can put this at the bottom of the document: 
<script>
    $(document).ready(function ()
    {
        var meter = new RGraph.Meter({
            id: 'cvs',
            min: 0,
            max: 10,
            value: 8,
            options: {
                centerx: 116,
                centery: 136,
                background: {
                    color: 'rgba(0,0,0,0)',
                    image: {
                        url: '/images/meter-sketch.png',
                        stretch: false,
                        offsetx: 5,
                        offsety: 6
                    }
                },
                needle: {
                    color: 'rgba(0,0,0,0)',
                    image: {
                        url: '/images/meter-sketch-needle.png',
                        offsetx: -5,
                        offsety: -5
                    }
                },
                colors: {
                    ranges: []
                },
                'tickmarks.big.num': 0,
                'tickmarks.small.num': 0,
                border: false,
                labels: false,
                centerpin: {
                    stroke: 'rgba(0,0,0,0)',
                    fill: 'rgba(0,0,0,0)'
                },
                angles: {
                    start: 2.7,
                    end: 6.52
                }
            }
        }).draw();
    
        meter.canvas.onclick = function (e)
        {
            var value = meter.getValue(e);
        
            if (typeof value === 'number') {
                meter.value = value;
                meter.grow();
            }
        }
    });
</script>
