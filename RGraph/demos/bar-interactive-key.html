An example of a Bar chart with an interactive key

The interactive key has been re-implemented as of June 2013 and added to more chart types.

[No canvas support]

This goes in the documents header (or you could place it just above the jQuery ready event code): 
<script src="http://ajax.googleapis.com/ajax/libs/jquery/1.10.2/jquery.min.js"></script>
<script src="RGraph.common.core.js"></script>
<script src="RGraph.common.dynamic.js"></script>
<script src="RGraph.common.key.js"></script>
<script src="RGraph.drawing.rect.js"></script>
<script src="RGraph.bar.js"></script>
Put this where you want the chart to show up: 
<canvas id="cvs" width="800" height="250" !style="border: 1px solid #ccc">
    [No canvas support]
</canvas>
This is the code that generates the chart. Because it's using the jQuery ready event you can put this at the bottom of the document: 
<script>
    $(document).ready(function ()
    {
        var bar1 = new RGraph.Bar({
            id: 'cvs',
            data: [[8,4,1],[4,8,8],[1,2,2],[3,3,4],[6,5,2],[5,8,8],[4,8,5]],
            options: {
                key: {
                    self: ['Jim','Jack','Joseph'],
                    interactive: true,
                    'text.size': 14
                },
                'background.grid.autofit.numvlines': 7,
                shadow: {
                    offsetx: 2,
                    offsety: -2,
                    blur: 5,
                    color: '#aaa'
                },
                labels: ['Monday','Tuesday','Wednesday','Thursday','Friday','Saturday','Sunday'],
                'text.size':14
            }
        }).draw()
    });
</script>
