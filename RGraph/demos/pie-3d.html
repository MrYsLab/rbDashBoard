A Pie chart with 3D effect (using gradients)

[No canvas support]

This goes in the documents header (or you could place it just above the jQuery ready event code): 
<script src="http://ajax.googleapis.com/ajax/libs/jquery/1.10.2/jquery.min.js"></script>
<script src="RGraph.common.core.js"></script>
<script src="RGraph.pie.js"></script>
Put this where you want the chart to show up: 
<canvas id="cvs" width="350" height="350">
    [No canvas support]
</canvas>
This is the code that generates the chart. Because it's using the jQuery ready event you can put this at the bottom of the document: 
<script>
    $(document).ready(function ()
    {
        var pie = new RGraph.Pie({
            id: 'cvs',
            data: [4,8,4,6,5,3,2,5],
            options: {
                gutter: {
                    left: 50,
                    right: 50
                },
                linewidth: 0,
                strokestyle: 'rgba(0,0,0,0)',
                labels: ['Alvin','Pete','Hoolio','Jack','Kev','Luis','Lou','Jesse'],
                colors: ['red','green','blue','cyan','yellow','pink','white','#ddd']
            }
        }).draw();

        // This Pie is a way of generating the gradient - the Pie chart is drawn on top of the first
        var pie2 = new RGraph.Pie({
            id: 'cvs',
            data: [1,0],
            options: {
                gutter: {
                    right: pie.gutterRight,
                    left: pie.gutterLeft
                },
                strokestyle: 'rgba(0,0,0,0)',
                colors: ['Gradient(rgba(148,148,148,0):rgba(148,148,148,0):rgba(148,148,148,0):rgba(148,148,148,0):rgba(148,148,148,0):rgba(148,148,148,0):rgba(0,0,0,0.25))']
            }
        }).draw();
    });
</script>
